//// -- LEVEL 1
//// -- Tables and References

// Main tables
Table users {
  userId serial [pk] // auto-serial
  userNickName varchar
  userName varchar
  userSurname varchar  
  userPhoneNumb int
  userEmail varchar
  userAdressStreet varchar
  userAdressNumber int
  userAdressApartment varchar
  userAdressCity varchar
  userAdressZipCode int [ref: > zipCode.zipCodeId]
  userAdressCountry int [ref: > country.countryId]
  userTaxCond int [ref: > taxCond.taxCondId]
}
Table products {
  productId serial [pk] // auto-serial
  productName varchar
  productDescription varchar
  productSize varchar
  productWeight varchar
  productStock int
  productPrice int
  productSeller int  [ref: > sellers.sellerId]
}
Table sellers {
  sellerId serial [pk]
  sellerNickName varchar
  sellerName varchar
  sellerSurname varchar  
  sellerPhoneNumb int
  sellerEmail varchar
  sellerAdressStreet varchar
  sellerAdressNumber int
  sellerAdressApartment varchar
  sellerAdressCity varchar
  sellerAdressZipCode int [ref: > zipCode.zipCodeId]
  sellerAdressCountry int [ref: > country.countryId]
  sellerTaxCond int [ref: > taxCond.taxCondId]
}

//// -- LEVEL 2
// Secondary tables
Table orders {
  orderId serial [pk]
  orderDate date
  orderUser int  [ref: > users.userId]
  orderState int  [ref: > states.stateId]
  orderPayment int  [ref: > payment.paymentId]
  orderDelivery int  [ref: > delivery.deliveryId]
}
Table items {
  itemId serial[pk]
  itemOrder int  [ref: > orders.orderId]
  itemProduct int  [ref: > products.productId]
  itemQuantity int
}
Table images {
  imageId serial [pk]
  imageProductId  int [ref: > products.productId]
  image varchar // ruta de la imagen
  imageOrder int // orden de muestra
}
Table states {
  stateId serial[pk]
  stateType varchar
}

//// -- LEVEL 3
// Support tables
Table country {
  countryId int [pk]
  countryName varchar
}
Table zipCode {
  zipCodeId int [pk]
  zipCodeLocation varchar
}
Table taxCond {
  taxCondId serial [pk]
  taxCondName varchar
}
Table payment {
  paymentId serial[pk]
  paymentType varchar
}
Table delivery {
  deliveryId serial [pk]
  deliveryType varchar
}
